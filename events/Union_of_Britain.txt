		trigger = {
			NOT = {

			}
		}
	}
	desc = {
		text = uob.16.d.return
		trigger = {
			OR = {
				GER = { has_country_flag = IRE_refused_protection }

			}
		}

		}

		log = "[GetDateText]: [Root.GetName]: event uob.16"

			}

			}
		}
	}

			factor = 40
		}

			}
