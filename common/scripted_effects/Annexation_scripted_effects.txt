syndicalist_socdem_government_change = {
	if = {
		limit = {
			ROOT = {
				has_government = social_democrat
				has_country_flag = soc_dems_are_socialists
			}
		}
		hidden_effect = {
			set_politics = {
				ruling_party = radical_socialist
				elections_allowed = no
			}
			add_popularity = {
				ideology = radical_socialist
				popularity = 0.4
			}
		}
	}
}
